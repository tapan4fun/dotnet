(

Table For Inpatient

create table hospital_inpatient
(
 patientname varchar(20) ,
 patientid int  IDENTITY(1001,1) primary key,
 gender varchar(20), 
 age int,
 address varchar(20),
 phoneres bigint,
 phonemob bigint,
 maritual varchar(20),
 occupation varchar(20),
 admid int ,
 admdate datetime,
 admtime datetime,
 status varchar(20),
 symtoms varchar(20),
 department varchar(20),
 wardno int,
 bedno int,
 doctor varchar(20)
);

--------------------------------------------------------------------------------------------------------------------------
Table For Patient Information

create table hospital_patientinfo
(
 pid int references hospital_inpatient(patientid),
 patientname varchar(20),
 age varchar(20),
 department varchar(20),
 doctor varchar(20)
);
